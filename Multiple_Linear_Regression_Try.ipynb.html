# Importing Libraries
import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.model_selection import train_test_split

# Load dataset
df = pd.read_csv('sample_data/california_housing_train.csv')
#%% 
# Function to calculate mean squared error
def mean_squared_error(y_true, y_predicted):
    cost = np.mean((y_predicted - y_true) ** 2)
    return cost

# Gradient Descent Function for Multiple Linear Regression
def gradient_descent(x, y, iterations, learning_rate, stopping_threshold):
    # Get sizes
    m , n = x.shape

    # Initialize Bias
    bias = 0.0

    # Initialize Weights
    weights = np.zeros(n)

    # Record Costs
    costs = []

    # Pre-Allocate previous cost
    cost_previous = None
    cost_curr = None

    # For each iteration
    for i in range(0,iterations):

        # Compute predictions
        y_predicted = np.dot(x,weights) + bias

        # Compute MSE
        cost_curr = mean_squared_error(y, y_predicted)


        # Check Threshold
        if cost_previous is not None and abs(cost_previous - cost_curr) <= stopping_threshold:
            break

        # Update Cost
        cost_previous = cost_curr
        costs.append(cost_curr)

        # Compute Gradient
        bias_grad = (-2/m) * np.sum(y - y_predicted)
        weight_grad =  (-2/m) * np.dot(x.T, (y - y_predicted))

        # Subtract weights by gradient
        bias -= learning_rate*bias_grad
        weights -= learning_rate*weight_grad

    # Plot cost
    plt.figure(figsize=(8, 6))
    plt.plot(range(len(costs)), costs)
    plt.title("Cost vs Iterations")
    plt.xlabel("Iterations")
    plt.ylabel("Cost")
    plt.show()


    return weights, bias, cost_curr

def main():
    # Select multiple features for x
    features = ['housing_median_age', 'total_bedrooms','population','median_income']

    # Create arrays
    x = df[features].values
    y = df['median_house_value'].values


    # Data Split
    x_train, x_test, y_train, y_test = train_test_split(x, y, random_state=349607)

    # Get mean and median (FOR X TRAIN ONLY, GET PARAMETERS FOR TRAIN THEN APPLY TO TRAIN AND TEST)
    x_mean = np.mean(x_train, axis=0)
    x_std = np.std(x_train, axis=0)

    # Normalize x_train
    x_train = (x_train-x_mean)/x_std

    # Normalize x_test
    x_test = (x_test-x_mean)/x_std

    # Call gradient_descent
    estimated_weights, estimated_bias, j = gradient_descent(x_train, y_train, 2000, 0.01, 0.000001)
    print('Bias Weight:', estimated_bias)
    print('Weights:', estimated_weights)
    print('Cost:', j)

    # Training: Make predictions
    y_pred_train = np.dot(x_train, estimated_weights) + estimated_bias

    # Training: plot predictions vs actual values
    plt.figure(figsize=(8, 6))
    plt.scatter(y_train, y_pred_train, color='blue', marker='o')
    plt.plot([min(y_train), max(y_train)], [min(y_train), max(y_train)], color='red', linestyle='dashed')
    plt.xlabel("Actual Values")
    plt.ylabel("Predicted Values")
    plt.title("Training: Actual vs Predicted")
    plt.show()

    mse_train = mean_squared_error(y_train, y_pred_train)
    print('MSE Train:', mse_train)

    # Test: Make predictions
    y_pred_test = np.dot(x_test, estimated_weights) + estimated_bias

    # Test: plot predictions vs actual values
    plt.figure(figsize=(8,6))
    plt.scatter(y_test, y_pred_test, color='blue', marker='o')
    plt.plot([min(y_test), max(y_test)], [min(y_test), max(y_test)], color ='red', linestyle='dashed')
    plt.xlabel('Actual Values')
    plt.ylabel('Predicted Values')
    plt.title('Test: Actual vs Predicted')
    plt.show()

    mse_test = mean_squared_error(y_test, y_pred_test)
    print('MSE Test:', mse_test)

if __name__ == "__main__":
    main()
